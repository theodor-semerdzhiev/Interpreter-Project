
func pow(x,n) {
    if(n == 0) {
        return 1;
    } else if(n == 1) {
        return x;
    }

    let i = 1;
    while(i < n) {
        x = x * x;
        i = i + 1;
    }
    return x;
}

func fact(n) {
    let res = 1;
    while(n > 0) {
        res = res * n;
        n = n - 1;
    }
    return res;
}

let PI = 3.1415926;

func sin(x) {
    let res = x;
    let term = x;
    let sign = -1;
    let i = 2;

    while(i < 20) {
        term = term * ((x * x) / ((i + 1) * i));
        res = res + sign * term;
        sign = sign * -1.0;
        i = i + 2;
    }
    return res;
}

func toRad(deg) {
    return (deg / 180) * PI;
}

print(sin(toRad(90)), "\n");
print(sin(toRad(60)), "\n");
