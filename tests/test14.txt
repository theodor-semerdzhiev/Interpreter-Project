class Tuple(list_) {
    let tuple = func () {
        if(typeof(list_) == "List") {
            return list_;
        } else {
            println("Argument must be a list");
            return [];
        }
    }();
    
    func printTuple() {

        print("(");
        for(let i=0; i < len(tuple); i = i + 1;) {
            print(" ", tuple[i]);
            if(!(i == len(tuple) -1)) {
                print(",");
            }
        } 
        print(")");
    }
}

let t = Tuple([1,3,4]);
t->printTuple();
